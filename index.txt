    <script>
        const SHEETS = {
            'Referencias': 'https://docs.google.com/spreadsheets/d/e/2PACX-1vSo_xmNEQffBWChyZawoj5YNVOhijLLGA-bpBGQHGiTyDeSKV_T-tdW1GQeBjsulx146kbq2V2SPXmN/pub?gid=491957809&single=true&output=csv',
            'Amorella': 'https://docs.google.com/spreadsheets/d/e/2PACX-1vSo_xmNEQffBWChyZawoj5YNVOhijLLGA-bpBGQHGiTyDeSKV_T-tdW1GQeBjsulx146kbq2V2SPXmN/pub?gid=1065082332&single=true&output=csv',
            'Emprenditoria': 'https://docs.google.com/spreadsheets/d/e/2PACX-1vSo_xmNEQffBWChyZawoj5YNVOhijLLGA-bpBGQHGiTyDeSKV_T-tdW1GQeBjsulx146kbq2V2SPXmN/pub?gid=1743770835&single=true&output=csv',
            'ID': 'https://docs.google.com/spreadsheets/d/e/2PACX-1vSo_xmNEQffBWChyZawoj5YNVOhijLLGA-bpBGQHGiTyDeSKV_T-tdW1GQeBjsulx146kbq2V2SPXmN/pub?gid=1231428952&single=true&output=csv',
            'Ginebra': 'https://docs.google.com/spreadsheets/d/e/2PACX-1vSo_xmNEQffBWChyZawoj5YNVOhijLLGA-bpBGQHGiTyDeSKV_T-tdW1GQeBjsulx146kbq2V2SPXmN/pub?gid=557549372&single=true&output=csv',
            'Benditta': 'https://docs.google.com/spreadsheets/d/e/2PACX-1vSo_xmNEQffBWChyZawoj5YNVOhijLLGA-bpBGQHGiTyDeSKV_T-tdW1GQeBjsulx146kbq2V2SPXmN/pub?gid=0&single=true&output=csv',
            'Taller': 'https://docs.google.com/spreadsheets/d/e/2PACX-1vSo_xmNEQffBWChyZawoj5YNVOhijLLGA-bpBGQHGiTyDeSKV_T-tdW1GQeBjsulx146kbq2V2SPXmN/pub?gid=753428539&single=true&output=csv',
            'Natalia': 'https://docs.google.com/spreadsheets/d/e/2PACX-1vSo_xmNEQffBWChyZawoj5YNVOhijLLGA-bpBGQHGiTyDeSKV_T-tdW1GQeBjsulx146kbq2V2SPXmN/pub?gid=1564348751&single=true&output=csv',
            'Manuela': 'https://docs.google.com/spreadsheets/d/e/2PACX-1vSo_xmNEQffBWChyZawoj5YNVOhijLLGA-bpBGQHGiTyDeSKV_T-tdW1GQeBjsulx146kbq2V2SPXmN/pub?gid=1147281710&single=true&output=csv'
        };

        let inventory = {};
        const LOCATIONS = Object.keys(SHEETS).filter(k => k !== 'Referencias');

        async function init() {
            try {
                const loads = Object.entries(SHEETS).map(([name, url]) => {
                    return new Promise(res => {
                        Papa.parse(url, {
                            download: true, header: true,
                            complete: results => { res({name, data: results.data}); },
                            error: () => { console.error("Error en " + name); res({name, data: []}); }
                        });
                    });
                });

                const allData = await Promise.all(loads);
                allData.forEach(item => process(item.name, item.data));
                
                document.getElementById('loader').style.display = 'none';
                document.getElementById('app').style.display = 'block';
                fillDatalist();
            } catch (e) {
                document.getElementById('errorBox').innerText = "Error cargando datos. Prueba refrescando.";
                document.getElementById('errorBox').style.display = 'block';
            }
        }

        function process(name, rows) {
            rows.forEach(row => {
                const ref = row['Referencia'];
                const talla = row['Talla'];
                if(!ref || !talla) return;
                if(!inventory[ref]) inventory[ref] = {};
                if(!inventory[ref][talla]) {
                    inventory[ref][talla] = {};
                    LOCATIONS.forEach(l => inventory[ref][talla][l] = 0);
                }
                if(name !== 'Referencias') {
                    let qty = parseInt(String(row['Saldo actual']).replace(/[^0-9]/g, '')) || 0;
                    inventory[ref][talla][name] = qty;
                }
            });
        }

        function fillDatalist() {
            const list = document.getElementById('refList');
            Object.keys(inventory).sort().forEach(r => {
                let opt = document.createElement('option');
                opt.value = r; list.appendChild(opt);
            });
        }

        document.getElementById('refInput').addEventListener('change', e => {
            const ref = e.target.value;
            const sel = document.getElementById('sizeSelect');
            sel.innerHTML = '<option value="ALL">Todas las tallas</option>';
            if(inventory[ref]) {
                Object.keys(inventory[ref]).forEach(t => {
                    let opt = document.createElement('option');
                    opt.value = t; opt.innerText = t; sel.appendChild(opt);
                });
                render(ref, 'ALL');
            }
        });

        document.getElementById('sizeSelect').addEventListener('change', e => {
            render(document.getElementById('refInput').value, e.target.value);
        });

        function render(ref, filter) {
            if(!inventory[ref]) return;
            let html = `<table><thead><tr><th>Talla</th>${LOCATIONS.map(l=>`<th>${l}</th>`).join('')}</tr></thead><tbody>`;
            const tallas = filter === 'ALL' ? Object.keys(inventory[ref]) : [filter];
            tallas.forEach(t => {
                html += `<tr><td>${t}</td>`;
                LOCATIONS.forEach(l => {
                    let q = inventory[ref][t][l];
                    html += `<td class="${q>0?'':'no-stock'}">${q > 0 ? q : '-'}</td>`;
                });
                html += `</tr>`;
            });
            html += '</tbody></table>';
            document.getElementById('tableResult').innerHTML = html;
        }

        init();
    </script>
